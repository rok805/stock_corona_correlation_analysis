def correlation(data):
    n=[]
    c=[]
    for i in company_name[:]:
        tmp = data[data[''].apply(lambda x: x==i)].reset_index(drop=True)
        c.append(tmp[['','']].corr().iloc[1,0])
        n.append(i)
        
    result = pd.DataFrame([n,c]).T
    
    result.columns = ['','']
    
    result = result.sort_values(by=[''], ascending=False).reset_index(drop=True)
    
    return result

def top_bar_graph(data,num=1):
    from collections import Counter

    tmp=pd.DataFrame([dict(Counter(data.)).keys(), dict(Counter(data.)).values()]).T
    tmp.columns=['','']
    tmp.sort_values(by='', ascending=False, inplace=True)
    tmp.reset_index(drop=True, inplace=True)

    bar=plt.bar(tmp., tmp., color='skyblue')
    bar[0].set_color('hotpink')
    bar[1].set_color('pink')

    plt.xticks(rotation=45, fontsize=15)
    plt.yticks(fontsize=15)
    plt.xlim(-1,10.5)
    if num ==1:
        plt.title('    ', fontsize= 20)
    else:
        plt.title('    ', fontsize= 20)
    plt.ylabel('', fontsize=15)
    plt.xlabel('', fontsize=15)
